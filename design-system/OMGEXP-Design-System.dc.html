Tokens as code

Drop these CSS custom properties into your root stylesheet. Names mirror the swatches above.

:root {
  /* brand */
  --color-navy-700: #184878;   /* primary */
  --color-navy-500: #2c6fac;   /* interactive */
  --color-green-500: #6fbe44;  /* action */
  --color-green-600: #5ea337;  /* action:hover */
  /* neutrals */
  --color-ink:       #16191c;
  --color-muted:     #5c656e;
  --color-border:    #e2e6ea;
  --color-surface:   #ffffff;
  --color-canvas:    #f8f9fa;
  /* type */
  --font-display: "Barlow", sans-serif;
  --font-body:    "Public Sans", sans-serif;
  /* scale (4px grid) */
  --space-1: 4px;  --space-2: 8px;  --space-3: 12px;
  --space-4: 16px; --space-6: 24px; --space-8: 32px;
  /* radius */
  --radius-sm: 2px; --radius-md: 4px; --radius-lg: 8px;
  /* elevation */
  --shadow-md: 0 4px 14px rgba(13,44,77,.10);
}
